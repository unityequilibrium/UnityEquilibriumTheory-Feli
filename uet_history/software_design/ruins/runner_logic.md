function step() {
  // 1. Get active equations
  const equations = getActiveEquations();
  
  // 2. Calculate forces
  for (each particle) {
    let force = { x: 0, y: 0, z: 0 };
    for (each equation) {
      force += equation.calculateForce(particle);
    }
  }
  
  // 3. Integrate (Verlet)
  for (each particle) {
    particle.velocity += force * dt;
    particle.position += particle.velocity * dt;
  }
  
  // 4. Update telemetry
  updateTelemetry();
  
  // 5. Increment step
  worldState.step += 1;
  worldState.time += dt;
}
